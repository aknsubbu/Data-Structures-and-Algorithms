def convertBST(root):
    if not root:
        return root
    global x
    x = 0
    newVal(root)
    return root


def newVal(node):
    global x
    if node.right:
        newVal(node.right)
    node.data += x
    x = node.data
    if node.left:
        newVal(node.left)

    start_time = perf_counter()

    func(bst)

def kthSmallest(root, k):
    nodes_in_order = []

    def traversal(node):
        if node is not None:
            traversal(node.left)
            nodes_in_order.append(node.data)
            traversal(node.right)

    traversal(root)
    return nodes_in_order[k - 1] if 0 <= k - 1 < len(nodes_in_order) else None

from random import randint
from time import perf_counter
import matplotlib.pyplot as plt
import tracemalloc


class Node:
    def __init__(self, data):
        self.data = data
        self.left = None
        self.right = None


def insert(root, data):
    if root is None:
        return Node(data)
    elif data <= root.data:
        root.left = insert(root.left, data)
    else:
        root.right = insert(root.right, data)
    return root


def generate_bst(n):
    root = None
    for _ in range(n):
        data = randint(1, NUMBER - 1)
        root = insert(root, data)
    return root


def lowestCommonAncestor(root, p, q):
    def findCommon(node, p, q):
        if node is None:
            return None
        if p > node.data and q > node.data:
            return findCommon(node.right, p, q)
        elif p < node.data and q < node.data:
            return findCommon(node.left, p, q)
        else:
            return node

    return findCommon(root, p, q)


def analyze_function(func, bst, p, q):
    start_time = perf_counter()
    tracemalloc.start()
    func(bst, p, q)
    end_time = perf_counter()
